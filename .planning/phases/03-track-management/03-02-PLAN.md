@mcp.tool()
def tool_name(ctx: Context, param: type = default) -> str:
    """Docstring with Parameters section."""
    try:
        ableton = get_ableton_connection()
        result = ableton.send_command("wire_name", {"param": param})
        return json.dumps(result, indent=2)
    except Exception as e:
        return format_error("Failed to ...", detail=str(e), suggestion="...")
