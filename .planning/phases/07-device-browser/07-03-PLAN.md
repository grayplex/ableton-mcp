from mcp.server.fastmcp import Context
from MCP_Server.connection import format_error, get_ableton_connection
from MCP_Server.server import mcp

@mcp.tool()
def tool_name(ctx: Context, param: type, optional_param: type | None = None) -> str:
    try:
        ableton = get_ableton_connection()
        params = {"key": value}
        if optional_param is not None:
            params["optional_key"] = optional_param
        result = ableton.send_command("command_name", params)
        return json.dumps(result, indent=2)
    except Exception as e:
        return format_error("Failed to ...", detail=str(e), suggestion="...")
