    **Step 2: Create MCP_Server/genres/schema.py with TypedDict definitions (per D-06)**

    Define these TypedDicts (per D-09, all structured data, no prose):

    ```python
    class InstrumentationSection(TypedDict):
        roles: List[str]  # Generic roles per D-11: "kick", "bass", "pad", "lead", etc.

    class HarmonySection(TypedDict):
        scales: List[str]  # Scale names matching SCALE_CATALOG keys (per D-10)
        chord_types: List[str]  # Chord quality names matching _QUALITY_MAP keys (per D-10)
        common_progressions: List[List[str]]  # Roman numeral sequences (per D-10)

    class RhythmSection(TypedDict):
        time_signature: str  # e.g., "4/4"
        bpm_range: List[int]  # [min, max] e.g., [120, 130]
        swing: str  # e.g., "none", "light", "heavy"
        note_values: List[str]  # e.g., ["1/4", "1/8", "1/16"]
        drum_pattern: str  # Brief description, e.g., "four-on-the-floor kick, offbeat hi-hats"

    class ArrangementEntry(TypedDict):
        name: str  # Section name, e.g., "intro"
        bars: int  # Number of bars

    class ArrangementSection(TypedDict):
        sections: List[ArrangementEntry]  # Per D-12: [{"name": "intro", "bars": 16}, ...]

    class MixingSection(TypedDict):
        frequency_focus: str  # e.g., "low-end emphasis with sub-bass"
        stereo_field: str  # e.g., "wide pads, mono bass"
        common_effects: List[str]  # e.g., ["sidechain compression", "reverb", "delay"]
        compression_style: str  # e.g., "heavy sidechain on bass/pads"

    class ProductionTipsSection(TypedDict):
        techniques: List[str]  # Genre-specific production techniques
        pitfalls: List[str]  # Common mistakes to avoid

    class GenreBlueprint(TypedDict):
        name: str  # Display name, e.g., "House"
        id: str  # Canonical identifier, e.g., "house"
        bpm_range: List[int]  # [min, max]
        aliases: List[str]  # Alternative names for alias resolution
        instrumentation: InstrumentationSection
        harmony: HarmonySection
        rhythm: RhythmSection
        arrangement: ArrangementSection
        mixing: MixingSection
        production_tips: ProductionTipsSection
    ```

    **Step 3: Create validate_blueprint() function (per D-07)**

    ```python
    def validate_blueprint(data: dict) -> None:
        """Validate a genre blueprint dict against required schema.

        Raises ValueError if data is malformed. Returns None on success.
        Per D-07: checks required keys and types at import time.
        Per D-08: caller (catalog) catches ValueError per genre.
        """
    ```

    Validation checks (in order):
    1. `data` must be a dict — raise ValueError("Blueprint must be a dict")
    2. Required top-level keys: `name`, `id`, `bpm_range`, `aliases`, `instrumentation`, `harmony`, `rhythm`, `arrangement`, `mixing`, `production_tips` — raise ValueError(f"Missing required key: {key}")
    3. `name` must be str, `id` must be str, `bpm_range` must be list of 2 ints, `aliases` must be list
    4. `instrumentation` must have `roles` as non-empty list of strings
    5. `harmony` must have `scales` (non-empty list of str), `chord_types` (non-empty list of str), `common_progressions` (non-empty list of lists)
    6. `rhythm` must have `time_signature` (str), `bpm_range` (list of 2 ints), `swing` (str), `note_values` (non-empty list), `drum_pattern` (str)
    7. `arrangement` must have `sections` as non-empty list, each entry must have `name` (str) and `bars` (int)
    8. `mixing` must have `frequency_focus` (str), `stereo_field` (str), `common_effects` (non-empty list), `compression_style` (str)
    9. `production_tips` must have `techniques` (non-empty list of str), `pitfalls` (non-empty list of str)

    Use standard library imports only: `from typing import TypedDict, List`

    ```python
    import pytest
    from MCP_Server.genres.schema import (
        GenreBlueprint,
        validate_blueprint,
    )

    def _make_valid_blueprint() -> dict:
        """Return a minimal valid blueprint dict for testing."""
        return {
            "name": "Test Genre",
            "id": "test_genre",
            "bpm_range": [120, 130],
            "aliases": ["test"],
            "instrumentation": {"roles": ["kick", "bass", "pad"]},
            "harmony": {
                "scales": ["minor"],
                "chord_types": ["min7"],
                "common_progressions": [["i", "iv"]],
            },
            "rhythm": {
                "time_signature": "4/4",
                "bpm_range": [120, 130],
                "swing": "none",
                "note_values": ["1/4", "1/8"],
                "drum_pattern": "four-on-the-floor",
            },
            "arrangement": {
                "sections": [{"name": "intro", "bars": 16}, {"name": "drop", "bars": 32}],
            },
            "mixing": {
                "frequency_focus": "low-end",
                "stereo_field": "wide pads, mono bass",
                "common_effects": ["sidechain compression"],
                "compression_style": "heavy sidechain",
            },
            "production_tips": {
                "techniques": ["layer kicks"],
                "pitfalls": ["muddy low-end"],
            },
        }

    class TestSchema:
        def test_schema_has_six_dimensions(self):
            # Verify GenreBlueprint annotations include all 6 dimensions
            annotations = GenreBlueprint.__annotations__
            for dim in ["instrumentation", "harmony", "rhythm", "arrangement", "mixing", "production_tips"]:
                assert dim in annotations, f"Missing dimension: {dim}"

        def test_validate_valid_blueprint(self):
            validate_blueprint(_make_valid_blueprint())  # should not raise

        def test_validate_missing_top_level_key(self):
            bp = _make_valid_blueprint()
            del bp["harmony"]
            with pytest.raises(ValueError, match="harmony"):
                validate_blueprint(bp)

        def test_validate_missing_section_key(self):
            bp = _make_valid_blueprint()
            del bp["harmony"]["scales"]
            with pytest.raises(ValueError, match="scales"):
                validate_blueprint(bp)

        def test_validate_wrong_type(self):
            bp = _make_valid_blueprint()
            bp["bpm_range"] = "fast"
            with pytest.raises(ValueError):
                validate_blueprint(bp)

        def test_validate_empty_section(self):
            bp = _make_valid_blueprint()
            bp["instrumentation"]["roles"] = []
            with pytest.raises(ValueError):
                validate_blueprint(bp)
    ```
